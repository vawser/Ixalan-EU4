# Ruler Death
#----------------------------------------------------
country_event = {
	id = mod_ruler.10
	title = mod_ruler.10.title
	desc = mod_ruler.10.desc
	picture = DEATH_OF_HEIR_eventPicture
    
	trigger = {
        has_regency = no
	    NOT = { is_immortal_lifespan = yes }
	}

    mean_time_to_happen = { 
        months = 1200 # 100
        
        modifier = {
            factor = 1.5
            
            is_increased_lifespan = yes
        }
        
        modifier = {
            factor = 2
            
            NOT = { ruler_age = 40 }
        }
        modifier = {
            factor = 0.5
            
            ruler_age = 100
        }
        modifier = {
            factor = 0.5
            
            ruler_age = 125
        }
        modifier = {
            factor = 0.5
            
            ruler_age = 150
        }
        modifier = {
            factor = 0.8
            
            is_monarch_leader = yes
        }
    }
    
	option = {
		name = mod_ruler.10.option.a
        
        kill_ruler = yes
	}
}
